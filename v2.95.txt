void obtenerDistancia() {
  digitalWrite(outputUltrasonido, LOW);
  delayMicroseconds(2);
  digitalWrite(outputUltrasonido, HIGH);
  delayMicroseconds(10);
  digitalWrite(outputUltrasonido, LOW);
  distanciaUltrasonido = (pulseIn(inputUltrasonido, HIGH)) / 58;
}

// DISTANCIAS
#define CUERPOACUERPO 0
#define CERCA 13
#define MEDIO 45
#define LEJOS 70
int distancia;

void calcularDistancia(){
  obtenerDistancia();
  if(distancia <= CERCA) distancia = CUERPOACUERPO;
  else if(distancia <= MEDIO) distancia = CERCA;
  else if(distancia <= LEJOS) distancia = MEDIO;
  else distancia = LEJOS;
}

// ESTADOS
#define CENTRO 0
#define CENTRODERECHA 1
#define DERECHA 2
#define CENTROIZQUIERDA 3
#define IZQUIERDA 4
int estado = CENTRO;

// POSICIONES DETECTADAS DEL CONTRINCANTE
/*
#define CENTRO 0
#define CENTRODERECHA 1
#define DERECHA 2
#define CENTROIZQUIERDA 3     TODO ESTO LO HEMOS DEFINIDO ANTES
#define IZQUIERDA 4
*/

// Estas son posiciones especiales: desconocida pero tras empezar el rastreo y desconocida pero al terminar el rastreo
#define DESCONOCIDA 5
#define DESCONOCIDAFIN 6
int posicion = DESCONOCIDA;

// VARIABLE PARA SABER HACIA DONDE HAY QUE 
int torcer = DERECHA;

// MODO LIGERO O TANQUE
#define LIGERO 0
#define TANQUE 1
int modo = LIGERO;

void loop(){
  maquinaEstados(); // POR MOTIVOS DE DEPURACIÓN, SE DECIDE IMPLEMENTAR CADA PALMO DE CÓDIGO ITERATIVO ENCAPSULADO EN UN PROCEDIMIENTO ALEJADO DE LOOP
}

void maquinaEstados(){
  // MEDIMOS LA DISTANCIA Y GUARDAMOS LO QUE VALE 
  calcularDistancia();

  // SI VEMOS AL CONTRINCANTE EN EL ESTADO ACTUAL, ENTONCES LA POSICIÓN DEL CONTRINCANTE PASA A SER EL ESTADO
  if(distancia != LEJOS){
    posicion = estado;
  } // EN CASO CONTRARIO, LA POSICIÓN PERMANECERÁ EN "DESCONOCIDA" O EN "DECONOCIDAFIN"
  
  // SELECCIÓN DEL ESTADO
  if(estado == CENTRO) estadoCentro();
//  else if(estado == CENTRODERECHA) estadoCentroDerecha();
  else if(estado == DERECHA) estadoDerecha();
  //else if(estado == CENTROIZQUIERDA) estadoCentroIzquierda();
  else if(estado == IZQUIERDA) estadoIzquierda();
}

void estadoCentro(){
  // A PARTIR DE LA POSICIÓN ACTUAMOS EN CONSECUENCIA
  if(posicion == DESCONOCIDA){
  // PREGUNTAMOS HACIA DONDE HABRÍA QUE GIRAR PARA LA BÚSQUEDA, YA QUE DESCONOCEMOS LA POSICIÓN DEL CONTRINCANTE
    if(torcer == DERECHA){
      torcerDer();
      estado = DERECHA;
    }
    if(torcer == IZQUIERDA){
      torcerIzq();
      estado = IZQUIERDA;
    }
  }
  else if(posicion == DESCONOCIDAFIN){
    // BASICAMENTE GIRAMOS A LA IZQUIERDA Y REPETIMOS TODO EL PROCESO DESDE EL CENTRO
      giroIzq();
      torcer = DERECHA;
      posicion = DESCONOCIDA;
      estado = CENTRO;   
  }
  else if(posicion == CUERPOACUERPO){
      // NADA POR AHORA
      estado = CENTRO;
  }
  else if(posicion == CERCA){
      modoTanque();
      delay(100);
      modo = TANQUE;
      // FALTARÍA QUE ANDASE HACIA DELANTE, PERO ANTES HAY QUE REALIZAR PRUEBAS PARCIALES A LA MÁQUINA DE ESTADOS
      estado = CENTRO;
  }
  else if(posicion == MEDIO){
      modo = LIGERO;
      levantar();
      bloqueBasicoMov;
      delay(300);
      bloqueBasicoMov();
      delay(300);
      reincorporacionTrasMov();
      estado = CENTRO;
  }
  
}

void estadoDerecha(){
   if(posicion == DERECHA){
      posicion = DESCONOCIDA; // OLVIDAMOS LA POSICIÓN (PARA QUE DESDE EL ESTADO CENTRO NO SE VUELVA A GIRAR A LA DERECHA PORQUE SÍ)
      estado = CENTRO;
      torcerCentro();
      giroDer();   
   }
   else{
      estado = CENTRO;
      torcerCentro();
      torcer = IZQUIERDA;
      posicion = DESCONOCIDA; // NO CREO QUE SE ACTUALICE DESDE EL VALOR "DESCONOCIDA", PERO LO PONGO POR SI ACASO
   }
}

void estadoIzquierda(){
   if(posicion == IZQUIERDA){
      posicion = DESCONOCIDA; // OLVIDAMOS LA POSICIÓN (PARA QUE DESDE EL ESTADO CENTRO NO SE VUELVA A GIRAR A LA DERECHA PORQUE SÍ)
      estado = CENTRO;
      torcerCentro();
      giroIzq();   
      torcer = DERECHA;
   }
   else{
      estado = CENTRO;
      torcerCentro();
      torcer = DERECHA;
      posicion = DESCONOCIDAFIN; // LO INDICAMOS PARA QUE DEJE DE BUSCAR Y GIRE HACIA LA IZQUIERDA
   }
}

void torcerCentro(){
  caderaFrente();
}

void torcerIzq(){
  pwm.setPWM(HOMBROIZQ, 0, 450 );
}

void torcerDer(){
  pwm.setPWM(CADERA, 0, 310);
}
